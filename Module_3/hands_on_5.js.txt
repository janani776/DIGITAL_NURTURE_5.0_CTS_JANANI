db.createCollection("feedback")

{

 rating:5
}
])

db.feedback.countDocuments()

// Task 2

db.feedback.find({rating:5})

db.feedback.updateMany(
 {rating:{$lt:3}},
 {$set:{needs_review:true}}
)

// Task 3

db.feedback.aggregate([
 {
   $group:{
      _id:"$course_code",
      avg_rating:{$avg:"$rating"}
   }
 }
])
